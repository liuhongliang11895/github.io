  <script>
    async function redirect() {
      try {
        const res = await fetch("mapping.json?ts=" + Date.now()); // 加随机参数避免缓存
        const data = await res.json();
        const activeLinks = data.links.filter(l => l.active);

        if (activeLinks.length === 0) {
          return; // 没有可用链接则停留
        }

        // 按权重抽签
        const weighted = [];
        activeLinks.forEach(l => {
          for (let i = 0; i < (l.weight || 1); i++) {
            weighted.push(l.url);
          }
        });
        const targetUrl = weighted[Math.floor(Math.random() * weighted.length)];

        // 立即跳转
        window.location.replace(targetUrl);
      } catch (e) {
        console.error("跳转失败", e);
      }
    }

    redirect();
  </script>
